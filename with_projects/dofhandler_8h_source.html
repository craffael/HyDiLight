442 class DynamicFEDofHandler : public DofHandler {
443  public:
467  template <typename LOCALDOFINFO>
468  DynamicFEDofHandler(std::shared_ptr<const lf::mesh::Mesh> mesh_p,
469  LOCALDOFINFO &&locdof)
470  : mesh_p_(std::move(mesh_p)) {
471  LF_ASSERT_MSG((mesh_p_->DimMesh() == 2), "Can handle 2D meshes only");
472 
473  // Index counter for global shape functions = global dof
474  gdof_idx_t dof_idx = 0;
475 
476  // Step I: Set indices for shape functions on nodes
477  // Run through node indices (entities of co-dimension 2)
478  const size_type no_nodes = mesh_p_->NumEntities(2);
479  num_int_dofs_[2].resize(no_nodes, 0);
480  offsets_[2].resize(no_nodes + 1, 0);
481  // Traverse nodes (co-dimension-2 entities) based on indices
482  for (glb_idx_t node_idx = 0; node_idx < no_nodes; node_idx++) {
483  // Obtain pointer to node entity
484  const mesh::Entity *node_p{mesh_p_->EntityByIndex(2, node_idx)};
485  LF_ASSERT_MSG(mesh_p_->Index(*node_p) == node_idx, "Node index mismatch");
486  // Offset for indices of node in index vector
487  glb_idx_t node_dof_offset = dof_idx;
488  offsets_[2][node_idx] = node_dof_offset;
489  // Request number of local shape functions associated with the node
490  size_type no_int_dof_node = locdof(*node_p);
491  num_int_dofs_[2][node_idx] = no_int_dof_node;
492 
493  // Store dof indices in array
494  for (unsigned j = 0; j < no_int_dof_node; j++) {
495  dofs_[2].push_back(dof_idx);
496  dof_entities_.push_back(node_p); // Store entity for current dof
497  dof_idx++; // Move on to next index
498  }
499  }
500  // Set sentinel
501  offsets_[2][no_nodes] = dof_idx;
502 
503  // Step II: Set indices for shape functions on edges (co-dimension = 1)
504  const size_type no_edges = mesh_p_->NumEntities(1);
505  // Set length of edge-related index vectors
506  num_int_dofs_[1].resize(no_edges, 0);
507  offsets_[1].resize(no_edges + 1, 0);
508  // points to the position of the current dof index
509  size_type edge_dof_offset = 0;
510  for (glb_idx_t edge_idx = 0; edge_idx < no_edges; edge_idx++) {
511  // Obtain pointer to edge entity
512  const mesh::Entity *edge{mesh_p_->EntityByIndex(1, edge_idx)};
513  LF_ASSERT_MSG(mesh_p_->Index(*edge) == edge_idx, "Edge index mismatch");
514  // Offset for indices of edge dof in index vector
515  offsets_[1][edge_idx] = edge_dof_offset;
516  size_type no_int_dof_edge = locdof(*edge);
517  num_int_dofs_[1][edge_idx] = no_int_dof_edge;
518 
519  // Obtain indices for basis functions sitting at endpoints
520  // Endpoints are mesh entities with co-dimension = 2
521  for (const lf::mesh::Entity *endpoint : edge->SubEntities(1)) {
522  const glb_idx_t ep_idx(mesh_p_->Index(*endpoint));
523  glb_idx_t ep_dof_offset = offsets_[2][ep_idx];
524  size_type no_int_dofs_ep = num_int_dofs_[2][ep_idx];
525  // Copy indices of shape functions from nodes to edge
526  for (unsigned j = 0; j < no_int_dofs_ep; j++) {
527  dofs_[1].push_back(dofs_[2][ep_dof_offset + j]);
528  edge_dof_offset++;
529  }
530  }
531  // Set indices for interior edge degrees of freedom
532  for (unsigned j = 0; j < no_int_dof_edge; j++) {
533  dofs_[1].push_back(dof_idx);
534  edge_dof_offset++;
535  dof_entities_.push_back(edge);
536  dof_idx++;
537  }
538  }
539  // Set sentinel
540  offsets_[1][no_edges] = edge_dof_offset;
541 
542  // Step III: Set indices for shape functions on cells (co-dimension = 0)
543  const size_type no_cells = mesh_p_->NumEntities(0);
544  // Set length of cell-related index vectors
545  num_int_dofs_[0].resize(no_cells, 0);
546  offsets_[0].resize(no_cells + 1, 0);
547  // points to the position of the current dof index
548  size_type cell_dof_offset = 0;
549  for (glb_idx_t cell_idx = 0; cell_idx < no_cells; cell_idx++) {
550  // Obtain pointer to current ell
551  const mesh::Entity *cell{mesh_p_->EntityByIndex(0, cell_idx)};
552  // Offset for indices of cell dof in index vector
553  offsets_[0][cell_idx] = cell_dof_offset;
554  size_type no_int_dof_cell = locdof(*cell);
555  num_int_dofs_[0][cell_idx] = no_int_dof_cell;
556 
557  // Obtain indices for basis functions in vertices
558  for (const lf::mesh::Entity *vertex : cell->SubEntities(2)) {
559  const glb_idx_t vt_idx(mesh_p_->Index(*vertex));
560  glb_idx_t vt_dof_offset = offsets_[2][vt_idx];
561  size_type no_int_dofs_vt = num_int_dofs_[2][vt_idx];
562  // Copy indices of shape functions from nodes to cell
563  for (unsigned j = 0; j < no_int_dofs_vt; j++) {
564  dofs_[0].push_back(dofs_[2][vt_dof_offset + j]);
565  cell_dof_offset++;
566  }
567  }
568 
569  // Collect indices of interior shape functions of edges
570  // Internal ordering will depend on the orientation of the edge
571  auto edge_orientations = cell->RelativeOrientations();
572  auto edges = cell->SubEntities(1);
573  // Loop over edges
574  for (int ed_sub_idx = 0; ed_sub_idx < cell->RefEl().NumSubEntities(1);
575  ed_sub_idx++) {
576  const glb_idx_t edge_idx = mesh_p_->Index(*edges[ed_sub_idx]);
577  const size_type no_int_dof_edge = num_int_dofs_[1][edge_idx];
578  const glb_idx_t edge_int_dof_offset =
579  offsets_[1][edge_idx + 1] - no_int_dof_edge;
580 
581  // Copy indices of shape functions from edges to cell
582  // The order, in which they are copied depends on the relative
583  // orientation of the edge w.r.t. the cell
584  switch (edge_orientations[ed_sub_idx]) {
585  case lf::mesh::Orientation::positive: {
586  for (int j = 0; j < no_int_dof_edge; j++) {
587  dofs_[0].push_back(dofs_[1][edge_int_dof_offset + j]);
588  cell_dof_offset++;
589  }
590  break;
591  }
592  case lf::mesh::Orientation::negative: {
593  for (int j = static_cast<int>(no_int_dof_edge - 1); j >= 0; j--) {
594  dofs_[0].push_back(dofs_[1][edge_int_dof_offset + j]);
595  cell_dof_offset++;
596  }
597  break;
598  }
599  } // end switch
600  } // end loop over edges
601 
602  // Set indices for interior shape functions of the cell
603  for (unsigned j = 0; j < no_int_dof_cell; j++) {
604  dofs_[0].push_back(dof_idx);
605  cell_dof_offset++;
606  dof_entities_.push_back(cell);
607  dof_idx++;
608  } // end loop over interior dof of cell
609  } // end loop over cells
610  // Set sentinel
611  offsets_[0][no_cells] = cell_dof_offset;
612 
613  // Finally set number of global shape functions
614  num_dof_ = dof_idx;
615  } // end constructor
