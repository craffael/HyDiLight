259 // Main method for the computation of the element matrix
260 template <typename SCALAR, typename DIFF_COEFF, typename REACTION_COEFF>
261 typename lf::uscalfe::ReactionDiffusionElementMatrixProvider<
262  SCALAR, DIFF_COEFF, REACTION_COEFF>::ElemMat
263 ReactionDiffusionElementMatrixProvider<
264  SCALAR, DIFF_COEFF, REACTION_COEFF>::Eval(const lf::mesh::Entity &cell) {
265  // Topological type of the cell
266  const lf::base::RefEl ref_el{cell.RefEl()};
267  // Obtain precomputed information about values of local shape functions
268  // and their gradients at quadrature points.
269  PrecomputedScalarReferenceFiniteElement<SCALAR> &pfe =
270  fe_precomp_[ref_el.Id()];
271  if (!pfe.isInitialized()) {
272  // Accident: cell is of a type not covered by finite element
273  // specifications or there is no quadrature rule available for this
274  // reference element type
275  std::stringstream temp;
276  temp << "No local shape function information or no quadrature rule for "
277  "reference element type "
278  << ref_el;
279  throw base::LfException(temp.str());
280  }
281 
282  // Query the shape of the cell
283  const lf::geometry::Geometry *geo_ptr = cell.Geometry();
284  LF_ASSERT_MSG(geo_ptr != nullptr, "Invalid geometry!");
285  LF_ASSERT_MSG((geo_ptr->DimLocal() == 2),
286  "Only 2D implementation available!");
287  SPDLOG_LOGGER_TRACE(reaction_diffusion_element_matrix_provider_logger,
288  "{}, shape = \n{}", ref_el,
289  geo_ptr->Global(ref_el.NodeCoords()));
290 
291  // Physical dimension of the cell
292  const dim_t world_dim = geo_ptr->DimGlobal();
293  // Gram determinant at quadrature points
294  const Eigen::VectorXd determinants(
295  geo_ptr->IntegrationElement(pfe.Qr().Points()));
296  LF_ASSERT_MSG(
297  determinants.size() == pfe.Qr().NumPoints(),
298  "Mismatch " << determinants.size() << " <-> " << pfe.Qr().NumPoints());
299  // Fetch the transformation matrices for the gradients
300  const Eigen::MatrixXd JinvT(
301  geo_ptr->JacobianInverseGramian(pfe.Qr().Points()));
302  LF_ASSERT_MSG(
303  JinvT.cols() == 2 * pfe.Qr().NumPoints(),
304  "Mismatch " << JinvT.cols() << " <-> " << 2 * pfe.Qr().NumPoints());
305  LF_ASSERT_MSG(JinvT.rows() == world_dim,
306  "Mismatch " << JinvT.rows() << " <-> " << world_dim);
307 
308  // compute values of coefficients alpha, gamma at quadrature points:
309  auto alphaval = alpha_(cell, pfe.Qr().Points());
310  auto gammaval = gamma_(cell, pfe.Qr().Points());
311 
312  // Element matrix
313  ElemMat mat(pfe.NumRefShapeFunctions(), pfe.NumRefShapeFunctions());
314  mat.setZero();
315 
316  // Loop over quadrature points
317  for (base::size_type k = 0; k < pfe.Qr().NumPoints(); ++k) {
318  const double w = pfe.Qr().Weights()[k] * determinants[k];
319  // Transformed gradients
320  const auto trf_grad(JinvT.block(0, 2 * k, world_dim, 2) *
321  pfe.PrecompGradientsReferenceShapeFunctions()
322  .block(0, 2 * k, mat.rows(), 2)
323  .transpose());
324  // Transformed gradients multiplied with coefficient
325  const auto alpha_trf_grad(alphaval[k] * trf_grad);
326  mat += w * (alpha_trf_grad.transpose() * trf_grad +
327  (gammaval[k] * pfe.PrecompReferenceShapeFunctions().col(k)) *
328  (pfe.PrecompReferenceShapeFunctions().col(k).transpose()));
329  }
330  return mat;
331 }

481  // Topological type of the cell
482  const lf::base::RefEl ref_el{edge.RefEl()};
483  LF_ASSERT_MSG(ref_el == lf::base::RefEl::kSegment(),
484  "Edge must be of segment type");
485  // Query the shape of the edge
486  const lf::geometry::Geometry *geo_ptr = edge.Geometry();
487  LF_ASSERT_MSG(geo_ptr != nullptr, "Invalid geometry!");
488 
489  // Obtain the metric factors for the quadrature points
490  const Eigen::VectorXd determinants(
491  geo_ptr->IntegrationElement(fe_precomp_.Qr().Points()));
492  LF_ASSERT_MSG(determinants.size() == fe_precomp_.Qr().NumPoints(),
493  "Mismatch " << determinants.size() << " <-> "
494  << fe_precomp_.Qr().NumPoints());
495 
496  // Element matrix
497  ElemMat mat(fe_precomp_.NumRefShapeFunctions(),
498  fe_precomp_.NumRefShapeFunctions());
499  mat.setZero();
500 
501  auto gammaval = gamma_(edge, fe_precomp_.Qr().Points());
502 
503  // Loop over quadrature points
504  for (long k = 0; k < determinants.size(); ++k) {
505  // Build local matrix by summing rank-1 contributions
506  // from quadrature points.
507  const auto w =
508  (fe_precomp_.Qr().Weights()[k] * determinants[k]) * gammaval[k];
509  mat += ((fe_precomp_.PrecompReferenceShapeFunctions().col(k)) *
510  (fe_precomp_.PrecompReferenceShapeFunctions().col(k).transpose())) *
511  w;
512  }
513  return mat;

860 ScalarLoadEdgeVectorProvider<SCALAR, FUNCTOR, EDGESELECTOR>::Eval(
861  const lf::mesh::Entity &edge) {
862  // Topological type of the cell
863  const lf::base::RefEl ref_el{edge.RefEl()};
864  LF_ASSERT_MSG(ref_el == lf::base::RefEl::kSegment(),
865  "Edge must be of segment type");
866  // Query the shape of the edge
867  const lf::geometry::Geometry *geo_ptr = edge.Geometry();
868  LF_ASSERT_MSG(geo_ptr != nullptr, "Invalid geometry!");
869 
870  // Quadrature points on physical edge
871  const Eigen::MatrixXd mapped_qpts(geo_ptr->Global(pfe_.Qr().Points()));
872  LF_ASSERT_MSG(
873  mapped_qpts.cols() == pfe_.Qr().NumPoints(),
874  "Mismatch " << mapped_qpts.cols() << " <-> " << pfe_.Qr().NumPoints());
875 
876  // Obtain the metric factors for the quadrature points
877  const Eigen::VectorXd determinants(
878  geo_ptr->IntegrationElement(pfe_.Qr().Points()));
879  LF_ASSERT_MSG(
880  determinants.size() == pfe_.Qr().NumPoints(),
881  "Mismatch " << determinants.size() << " <-> " << pfe_.Qr().NumPoints());
882 
883  // Element vector
884  ElemVec vec(pfe_.NumRefShapeFunctions());
885  vec.setZero();
886 
887  auto g_vals = g_(edge, pfe_.Qr().Points());
888 
889  // Loop over quadrature points
890  for (base::size_type k = 0; k < pfe_.Qr().NumPoints(); ++k) {
891  // Add contribution of quadrature point to local vector
892  const auto w = (pfe_.Qr().Weights()[k] * determinants[k]) * g_vals[k];
893  vec += pfe_.PrecompReferenceShapeFunctions().col(k) * w;
894  }
895  return vec;
896 }
